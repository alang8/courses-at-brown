3   import java.util.ArrayList;
4   import java.util.List;

6   public class Node implements Coordinate {
7     private static final double RADIUS = 6371.0;
8     private ArrayList<Double> coords = new ArrayList<>();
9     private String id;
10  
11    public Node(String id, double lat, double lon) {
12      this.coords.add(lat);
13      this.coords.add(lon);
14      this.id = id;
15    }
16  
17    @Override
18    public ArrayList<Double> getCoords() {
19      return coords;
20    }
21  
22    @Override
23    public int getID() {
24      return Integer.parseInt(id.substring(3).replaceAll("[.]", ""));
25    }
26  
27    @Override
28    public String getName() {
29      return this.id;
30    }
31  
32    /**
33     * Method which returns the haversine distance from another node.
34     * @param other - the other node to measure the distance from.
35     * @return - the distance from that node.
36     */
37    public double distanceFrom(List<Double> other) {
38      double sum = 0;
39      for (int i = 0; i < other.size(); i++) {
40        sum += Math.pow((this.coords.get(i) - other.get(i)), 2);
41      }
42      return Math.sqrt(sum);
43    }
44  
45    /**
46     *  Method which returns the haversine distance b/t two points.
47     * @param other - the lat and longitude of another Node.
48     * @return - the haversine distnace between this node and other.
49     */
50    public double haversineDistance(List<Double> other) {
51      double lat1 = this.coords.get(0);
52      double lon1 = this.coords.get(1);
53      double lat2 = other.get(0);
54      double lon2 = other.get(1);
55      double avgLat = (lat2 - lat1) / 2;
56      double avgLon = (lon2 - lon1) / 2;
57      double dis = Math.sin(avgLat) + Math.cos(lat1) * Math.cos(lat2) * Math.pow(Math.sin(avgLon), 2);
58      dis = Math.sqrt(dis);
59      dis = 2 * RADIUS * Math.asin(dis);
60      return dis;
61    }
62  
63    public String toString() {
64      return this.getName();
65    }

55    /**
56     * Method which builds a kd tree.
57     * @param sL - the list of objects to build the tree out of.
58     * @param dim - the dimensionality of the kd tree.
59     * @param depth - the current depth of the kd tree.
60     * @return - a kd tree filled with the objects in sL.
61     */
62    public static IKDTree buildTree(List<? extends Coordinate> sL, int dim, int depth) {
63      if (sL.isEmpty()) {
64        return new KDNode(null, null, null);
65      }
66  
67      int curAx = depth % dim;
68  
69      sL.sort(Comparator.comparingDouble(s -> s.getCoords().get(curAx)));
70      int listSz = sL.size();
71      int midIndex = listSz / 2;
72      Coordinate root = sL.get(midIndex);
73  
74      List<Double> curCoors = root.getCoords();
75  
76      //While loop so all "equals" are on the right
77      if (midIndex - 1 >= 0) {
78        boolean hasNext = true;
79        while (hasNext && (curCoors.get(curAx).equals(sL.get(midIndex - 1).getCoords().get(curAx)))) {
80          curCoors = sL.get(midIndex - 1).getCoords();
81          midIndex -= 1;
82          hasNext = (midIndex - 1 >= 0);
83        }
84      }
85  
86      root = sL.get(midIndex);
87  
88      List<Coordinate> leftSubList = (List<Coordinate>) sL.subList(0, midIndex);
89      List<Coordinate> rightSubList = (List<Coordinate>) sL.subList(midIndex + 1, listSz);
90  
91      IKDTree leftSubTree = buildTree(leftSubList, dim, depth + 1);
92      IKDTree rightSubTree = buildTree(rightSubList, dim, depth + 1);
93      return new KDNode(root, leftSubTree, rightSubTree);
94    }

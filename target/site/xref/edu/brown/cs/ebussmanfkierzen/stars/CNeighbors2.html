9   /**
10   * Class representing the neighbors command.
11   * @param <T> - the object w/ coordinates we are considering for neighbors.
12   */
13  public class CNeighbors2<T extends Coordinate> implements Command {
14  
15    private List<Coordinate> neighborList;
16    private int k;
17    private IKDTree<Coordinate> tree;
18    private List<String> results;
19    private List<Double> target;
20    private String targetName;
21    private Application app;
22  
23    /**
24     * execute: method which finds the n nearest neighbors to a given origin star.
25     *
26     * @param inputs an array of search criteria inputted by user
27     * @param appl   the application holding the data the user wishes to search through
28     * @return A list of strings representing the Star objects that are nearest
29     * to the origin and fit the requirements of "inputs"
30     **/
31    public List<String> execute(String[] inputs, Application appl) {
32      boolean searchByName = false;
33      this.app = appl;
34      this.target = new ArrayList<>();
35      this.results = new ArrayList<>();
36      if (!app.isDataLoaded()) {
37        System.out.println("ERROR: no loaded data");
38        results.add("ERROR: no loaded data");
39        return results;
40      } else if (!(inputs.length == 5 || inputs.length == 3)) {
41        System.out.println("ERROR: incorrect number of arguments for naive_neighbors");
42        results.add("ERROR: incorrect number of arguments for naive_neighbors");
43        return results;
44      } else if (Integer.parseInt(inputs[1]) < 0) {
45        System.out.println("ERROR: neighbors must be non-negative");
46        results.add("ERROR: neighbors must be non-negative");
47        return results;
48      }
49      if (inputs.length == 5) {
50        try {
51          target.add(Double.parseDouble(inputs[2]));
52          target.add(Double.parseDouble(inputs[3]));
53          target.add(Double.parseDouble(inputs[4]));
54        } catch (Exception e) {
55          System.out.println("ERROR: coordinates must be numeric values");
56          results.add("ERROR: coordinates must be numeric values");
57          return results;
58        }
59        targetName = "nothing";
60      } else {
61        Coordinate coord = app.findTargetByName(inputs[2]);
62        if (coord != null) {
63          searchByName = true;
64          this.targetName = coord.getName();
65          target = coord.getCoords();
66        } else {
67          System.out.println("ERROR: cannot find this star by name");
68          results.add("ERROR: cannot find this star by name");
69          return results;
70        }
71      }
72  
73      this.k = Integer.parseInt(inputs[1]);
74      this.tree = this.app.getData();
75      Comparator<Coordinate> myDoubleComp = Comparator.comparingDouble(s -> s.distanceFrom(target));
76      Comparator<Coordinate> starComp = Collections.reverseOrder(myDoubleComp);
77      PriorityQueue<Coordinate> pq = new PriorityQueue<Coordinate>(k + 1, starComp);
78      KDNode.kdnAlg(pq, this.tree, target, 0, k + 1);
79  
80      List<Coordinate> neighbors = new ArrayList(Arrays.asList(pq.toArray()));
81  
82      if (searchByName) {
83        neighbors.remove(app.findTargetByName(this.targetName));
84      }
85  
86      Collections.shuffle(neighbors);
87      neighbors.sort(Comparator.comparingDouble(s -> s.distanceFrom(target)));
88  
89      neighbors = new ArrayList(neighbors.subList(0, Math.min(k, neighbors.size())));
90  
91      for (Coordinate item : neighbors) {
92        System.out.println(item.getID());
93        results.add(item.toString());
94      }
95      this.neighborList = neighbors;
96      return results;
97    }
98

101     this.target = coords;
102   }

105   public List<Coordinate> getResults() {
106     return this.neighborList;
107   }
108 
109   @Override
110   public String forApp() {
111     return "stars";
112   }

9   /**
10   * class for handling parsing of CSV files.
11   */
12  public class CSVParse {
13  
14    public CSVParse() {
15    }
16  
17    /**
18     * csvParse: attempts to parse CSV files and returns packaged row information if possible.
19     *
20     * @param filename the name of the CSV file to be parsed
21     * @return a list of lists, where each element in the outer list is a row in the CSV file
22     * and each element in each inner list is a property or field of the CSV file
23     */
24    public List<String[]> csvParse(String filename) {
25  
26      try {
27        BufferedReader br = new BufferedReader(new FileReader(filename));
28        List<String[]> list = new ArrayList<>();
29        String line = br.readLine();
30        if (line.equals("first_name,last_name,date_time,email,gender,street_address") || line.equals("StarID,ProperName,X,Y,Z")) {
31          while ((line = br.readLine()) != null) {
32  
33            String[] attr = line.split(",", -1);
34            list.add(attr);
35          }
36          return list;
37        } else {
38          System.out.println("ERROR: improper file header");
39          return null;
40        }
41      } catch (IOException e) {
42        System.out.println("ERROR: file does not exist");
43        return null;
44      }
45    }
46  }

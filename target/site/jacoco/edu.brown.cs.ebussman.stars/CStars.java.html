package edu.brown.cs.ebussman.stars;

import java.util.List;
import java.util.ArrayList;

/**
 * CStars: handles functionality related to loading stars from a CSV file into an
 * instance of the application class.
 */
public class CStars implements Command {

  private List<String> results;
  private Application app;
  private List<Double> target;

  public CStars() {
  }

  public CStars(Application appl) {
    this.app = appl;
  }

  /**
   * execute: handles execution of "stars" command, including list of stars and
   * kd tree of stars, and throws any errors if applicable.
   *
   * @param inputs String array representing parsed command lind inputs
   * @param appl   stars application that will hold the data processed in this
   * @return a List of Strings containing an informative message about the status of the
   * data that was attempted to be loaded
   */
  public List<String> execute(String[] inputs, Application appl) {

    this.app = appl;
    this.results = new ArrayList<>();
    if (inputs.length != 2) {
      System.out.println("ERROR: improper arguments for stars command");
      results.add("ERROR: improper arguments for stars command");
      return results;
    } else {
      boolean loadedWithoutError = loadStars(inputs[1]);
      if (loadedWithoutError) {
        //build KD Tree
        app.setData(new KDTree<Coordinate>(app.getItemList(), 2, 2));
        results.add("Loaded Stars Successfully!");
        return results;
      } else {
        results.add("ERROR: malformed CSV file");
        return results;
      }
    }
  }

  /**
   * getStars method: uses CSVParse method to organize collection of Stars objects from a CSV file.
   *
   * @param filename the name of the CSV file containing rows representing stars
   * @return true if stars CSV file was successfully loaded without error, false otherwise
   */
  public boolean loadStars(String filename) {

    //overwrite old stars list if it exists
    List<Coordinate> starList = new ArrayList<>();
    CSVParse parser = new CSVParse();
    List<String[]> newStars = parser.csvParse(filename);

    if (newStars != null) {
      int starsRead = 0;
      for (String[] item : newStars) {
        if (item.length != 5) {
          System.out.println("ERROR: malformed CSV file");
          return false;
        }
        Star newStar = new Star(Integer.parseInt(item[0]), item[1], Double.parseDouble(item[2]), Double.parseDouble(item[3]), Double.parseDouble(item[4]));
        starList.add(newStar);
        starsRead++;
      }
      System.out.println("Read " + starsRead + " stars from " + filename);
      app.setDataLoaded();
      this.app.setItemList(starList);
      return true;
    }
    return false;
  }

  public void setTarget(List<Double> target) {
    this.target = target;
  }

  public List<Coordinate> getResults() {
    return null;
  }
}

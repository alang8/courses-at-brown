package edu.brown.cs.ebussman.stars;

/**
 * class representing a hypothetical person.
 */
public class MockPerson {

  private String firstName;
  private String lastName;
  private String dateTime;
  private String email;
  private String gender;
  private String streetAddress;

  public MockPerson(String firstName, String lastName, String dateTime, String email, String gender,
                    String streetAddress) {
    this.firstName = firstName;
    this.lastName = lastName;
    this.dateTime = dateTime;
    this.email = email;
    this.gender = gender;
    this.streetAddress = streetAddress;
  }

  @Override
  public String toString() {
    if (firstName.equals("") || firstName.equals(" ")) {
      System.out.println("ERROR: first name should be non-empty");
      firstName = "N/A";
    }
    if (lastName.equals("") | lastName.equals(" ")) {
      System.out.println("ERROR: last names should be non-empty");
      lastName = "N/A";
    }
    if (gender.equals("") || gender.equals(" ")) {
      gender = "N/A";
    }
    if (!email.matches(".*@.*..*") || email.equals("")) {
      System.out.println("ERROR: improperly formatted email");
      email = "N/A";
    }
    if (streetAddress.equals("")) {
      streetAddress = "N/A";
    }
    if (dateTime.equals("")) {
      dateTime = "N/A";
    }
    return "Name: " + firstName + " " + lastName + ", Address: " + streetAddress + ", Email: "
      + email + ", Gender: " + gender + ", Date/time: " + dateTime + "\n";
  }
}

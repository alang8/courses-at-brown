package edu.brown.cs.ebussman.stars;

import java.util.List;

/**
 * StarsApp class: holds star-related data and functionality
 * common to multiple stars commands.
 */
public class StarsApp<T extends Coordinate> implements Application {

  private List<Coordinate> itemList;
  private Coordinate target;
  private String[] inputs;
  private boolean dataLoaded;
  private KDTree<Coordinate> tree;
  private List<Star> results;
  private int dimension;


  public StarsApp(int dimension) {
    this.dimension = dimension;
    this.results = null;
  }


  /**
   * distance method: calculates the Euclidean distance between an input star and the origin star.
   *
   * @param a star from which to calculate distance
   * @param b star from which to calculate distance
   * @return double representing the distance between the two stars
   */

  public double distance(List<Double> a, List<Double> b) {

    double diff = 0.0;

    for (int i = 0; i < a.size(); i++) {
      diff += (a.get(i) - b.get(i)) * (a.get(i) - b.get(i));
    }
    return Math.sqrt(diff);
  }

  /**
   * findStarByName: given an list of stars, finds the star corresponding to a given input name, if
   * it exists, and sets corresponding fields accordingly.
   *
   * @param targetName name of target Star to be searched for
   * @return a Star matching the name inputted by the user
   */
  public Coordinate findTargetByName(String targetName) {

    // gets rid of quotation marks in star name
    String name = targetName.replaceAll("^\"|\"$", "");

    if (name.equals("")) {
      System.out.println("ERROR: Cannot search for nearest neighbors with empty name");
      return null;
    }
    // need to find star with this name
    int i = 0;
    while (i < itemList.size()) {
      if (itemList.get(i).getName().equals(name)) {
        Coordinate match = itemList.get(i);
        List<Double> coords = match.getCoords();
        return itemList.get(i);
      } else {
        i++;
      }
    }
    // match wasn't found
    System.out.println("ERROR: cannot find this star by name");
    return null;
  }


  public boolean isDataLoaded() {
    return this.dataLoaded;
  }

  public List<Coordinate> getItemList() {
    return itemList;
  }

  public int getDimension() {
    return this.dimension;
  }

  public KDTree<Coordinate> getData() {
    return this.tree;
  }

  public void setData(KDTree<Coordinate> tree1) {
    this.tree = tree1;
  }

  public String[] getArgs() {
    return this.inputs;
  }

  public void setArgs(String[] inputs1) {
    this.inputs = inputs1;
  }

  public void setItemList(List<Coordinate> items) {
    this.itemList = items;
  }

  public void setDataLoaded() {
    this.dataLoaded = true;
  }
}

package edu.brown.cs.ebussman.stars;

import java.util.Comparator;

/**
 * dynamic comparator: class allowing comparison of two
 * items on the basis of a specific axis or dimension.
 */
public class DynamicComparator<T extends Star> implements Comparator<T> {

  //value specifying the axis on which to compare two stars' coordinates
  //in this context, 0 cooresponds to x-axis, 1 to y-axis, and 2 to z-axis
  private int compareBy;

  public DynamicComparator(int compareBy) {
    this.compareBy = compareBy;
  }

  public DynamicComparator<T> compareBy(int compareVal) {
    this.compareBy = compareVal;
    return this;
  }

  /**
   * compare: compares to Coordinates.
   *
   * @param a : Coordinate object to be compared
   * @param b : Coordinate object to be compared
   * @return 0 if a and b have the same value on the given axis,
   * 1 if a is greater on specific axis and -1 if a is
   * less on the specified axis
   */
  public int compare(Star a, Star b) {
    return Double.compare(a.getCoords().get(this.compareBy), b.getCoords().get(this.compareBy));
  }

}

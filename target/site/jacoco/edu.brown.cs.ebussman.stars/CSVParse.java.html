package edu.brown.cs.ebussman.stars;

import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

/**
 * class for handling parsing of CSV files.
 */
public class CSVParse {

  public CSVParse() {
  }

  /**
   * csvParse: attempts to parse CSV files and returns packaged row information if possible.
   *
   * @param filename the name of the CSV file to be parsed
   * @return a list of lists, where each element in the outer list is a row in the CSV file
   * and each element in each inner list is a property or field of the CSV file
   */
  public List<String[]> csvParse(String filename) {

    try {
      BufferedReader br = new BufferedReader(new FileReader(filename));
      List<String[]> list = new ArrayList<>();
      String line = br.readLine();
      if (line.equals("first_name,last_name,date_time,email,gender,street_address") || line.equals("StarID,ProperName,X,Y,Z")) {
        while ((line = br.readLine()) != null) {

          String[] attr = line.split(",", -1);
          list.add(attr);
        }
        return list;
      } else {
        System.out.println("ERROR: improper file header");
        return null;
      }
    } catch (IOException e) {
      System.out.println("ERROR: file does not exist");
      return null;
    }
  }
}

package edu.brown.cs.ebussman.stars;

import java.util.List;
import java.util.Comparator;
import java.util.Collections;

/**
 * KDTree class: recursively creates a k-dimensional tree of the given type.
 */
public class KDTree<T extends Coordinate> {

  //this is the "node"
  private T val;
  private int depth;
  private KDTree<T> left;
  private KDTree<T> right;
  private List<Comparator<T>> comparators;


  // list of n-dimensional coordinates, current dimension/axis, max dimension
  public KDTree(List<T> coords, int depth, int maxDepth) {

    //base cases
    if (coords.size() == 1) {
      this.val = coords.get(0);

      this.left = null;
      this.right = null;
    } else {

      //find splitting axis
      if (depth == -1) {
        depth = maxDepth;
      }

      // sort coordinates by axis (ie. by the depth-th element of getCoords)
      DynamicComparator c = new DynamicComparator(depth);
      Collections.sort(coords, c);

      // get median, set val
      // list size is even
      int length = coords.size();
      int middleElem;

      // length is even
      if (length % 2 == 0) {
        middleElem = (length / 2) - 1;
      } else {
        // length is odd
        middleElem = (length - 1) / 2;
      }
      //checking for duplicate values near middle, always choose leftmost duplicate for consistency
      if (middleElem != 0 && coords.get(middleElem).getCoords().get(depth) == coords.get(middleElem - 1).getCoords().get(depth)) {
        //System.out.println("loop");
        while (coords.get(middleElem).getCoords().get(depth) == coords.get(middleElem - 1).getCoords().get(depth)) {
          middleElem--;
        }
      }

      int upper = middleElem;
      this.val = coords.get(middleElem);

      //adjust middle element for 0-indexing
      if (middleElem == 0) {
        upper++;
      }

      //one leftover element, choose to put it on right subtree
      if (coords.size() == 2) {
        this.left = null;
        this.right = new KDTree<>(coords.subList(middleElem + 1, length), depth - 1, maxDepth);
      } else {
        // recur on left/right subtree
        this.left = new KDTree<>(coords.subList(0, upper), depth - 1, maxDepth);
        this.right = new KDTree<>(coords.subList(middleElem + 1, length), depth - 1, maxDepth);
      }
    }
  }

  /**
   * @return left subtree of the given KDTree
   */
  public KDTree<T> getLeft() {
    return this.left;
  }

  /**
   * @return right subtree of the given KDTree
   */
  public KDTree<T> getRight() {
    return this.right;
  }

  /**
   * @return value of root node of the given KDTree
   */
  public T getVal() {
    return this.val;
  }

}

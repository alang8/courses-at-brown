package edu.brown.cs.ebussman.stars;

import java.util.Comparator;
import java.util.List;

/**
 * Star comparator: provides a means by which to compare stars based on their relative distance
 * to a given origin star.
 */
public class CoordComparator implements Comparator<Coordinate> {

  private Application app;
  private List<Double> target;

  public CoordComparator(Application app, List<Double> target) {
    this.app = app;
    this.target = target;
  }

  /**
   * compare: compares two stars on the basis of distance from a target.
   *
   * @param a: first coordinate to be compared
   * @param b: second coordinate to be compared
   * @return 0 if the coordinates are equidistant from the target coordinate
   * 1, if a is further away from the target, and -1 if a is closer to the target
   */
  public int compare(Coordinate a, Coordinate b) {
    double distanceA = app.distance(a.getCoords(), target);
    double distanceB = app.distance(b.getCoords(), target);
    return Double.compare(distanceA, distanceB);
  }

}

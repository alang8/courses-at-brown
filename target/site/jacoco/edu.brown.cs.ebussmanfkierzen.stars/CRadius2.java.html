package edu.brown.cs.ebussmanfkierzen.stars;

import java.util.LinkedList;
import java.util.Map;
import java.util.List;
import java.util.Iterator;
import java.util.ArrayList;
import java.util.HashMap;

/**
 * CRadius class: class used for functionality related
 * to finding items within a specified radius of a target.
 */
public class CRadius2<T extends Coordinate> implements Command {

  private int maxDepth;
  private LinkedList<Coordinate> radiusList;
  private int goalRadius;
  private Map<Coordinate, Double> neighborDistances;
  private IKDTree<Coordinate> tree;
  private List<Double> target;
  private Application app;
  private String targetName;

  public CRadius2() {
  }

  /**
   * execute: finds the specified number of elements within a given radius of a target
   * using an advanced, n-dimensional tree-based algorithm.
   *
   * @param inputs the search criteria for the radius search (name, radius value, etc)
   * @param appl   the application holding the data the user wishes to search through
   * @return a list of strings representing items within a given radius of a target
   */
  public List<String> execute(String[] inputs, Application appl) {

    this.radiusList = new LinkedList<>();
    this.app = appl;
    List<String> results = new ArrayList<>();
    this.target = new ArrayList<>();
    // do all error-checking inside this function
    if (!app.isDataLoaded()) {
      System.out.println("ERROR: no loaded data");
      results.add("ERROR: no loaded data");
      return results;
    } else if (!(inputs.length == 5 || inputs.length == 3)) {
      System.out.println("ERROR: incorrect number of arguments for naive_radius");
      results.add("ERROR: incorrect number of arguments for naive_radius");
      return results;
    }

    try {
      this.goalRadius = Integer.parseInt(inputs[1]);
    } catch (Exception e) {
      System.out.println("ERROR: radius must be non-negative");
      results.add("ERROR: radius must be non-negative");
      return results;
    }

    if (inputs.length == 5) {
      try {
        target.add(Double.parseDouble(inputs[2]));
        target.add(Double.parseDouble(inputs[3]));
        target.add(Double.parseDouble(inputs[4]));
      } catch (Exception e) {
        System.out.println("ERROR: coordinates must be numeric values");
        results.add("ERROR: coordinates must be numeric values");
        return results;
      }
      targetName = null;
    } else {
      Coordinate coord = app.findTargetByName(inputs[2]);
      if (coord != null) {
        target = coord.getCoords();
        this.targetName = coord.getName();
      } else {
        System.out.println("ERROR: cannot find this star by name");
        results.add("ERROR: cannot find this star by name");
        return results;
      }
    }

    this.tree = app.getData();
    this.neighborDistances = new HashMap<Coordinate, Double>();
    this.app = appl;
    this.maxDepth = app.getDimension() - 1;

    radius(this.tree, maxDepth);

    for (Coordinate item : radiusList) {
      System.out.println(item.getID());
      results.add(item.toString());
    }

    return results;
  }

  /**
   * radius: recursive algorithm that finds all items within a given radius of the current data (tree).
   *
   * @param currentTree the current KD-tree through which to search for
   *                    *               items within a radius
   * @param depth       the current axes on which to compare coordinates
   *                    *             (x is depth 0, y is depth 1, z is depth 2, for example)
   */
  public void radius(IKDTree<Coordinate> currentTree, int depth) {
    if (currentTree == null || currentTree.getVal() == null) {
      return;
    }

    if (depth == -1) {
      depth = this.maxDepth;
    }

    double straightDist = app.distance(currentTree.getVal().getCoords(), target);

    if (targetName != null && !targetName.equals(currentTree.getVal().getName())) {
      if (straightDist <= goalRadius) {

        neighborDistances.put(currentTree.getVal(), straightDist);

        if (radiusList.size() == 0 || straightDist > neighborDistances.get(radiusList.getLast())) {
          radiusList.add(currentTree.getVal());
        } else {
          addToList(currentTree.getVal(), straightDist);
        }
      }
    }

    double axialDist = Math.abs(target.get(depth) - currentTree.getVal().getCoords().get(depth));

    if (axialDist <= goalRadius) {
      radius(currentTree.getLeft(), depth - 1);
      radius(currentTree.getRight(), depth - 1);
    } else {

      if (target.get(depth) >= currentTree.getVal().getCoords().get(depth)) {
        radius(currentTree.getRight(), depth - 1);
      } else {
        radius(currentTree.getLeft(), depth - 1);
      }
    }
  }

  /**
   * addToList: adds a coordinate to radius list in the correct (sorted) position.
   *
   * @param node the current node to be added to the list of neighbors
   * @param nodeDist the distance from the current node to the target node
   */
  public void addToList(Coordinate node, Double nodeDist) {

    boolean added = false;
    Iterator<Coordinate> it = radiusList.iterator();
    int i = 0;

    while (!added && it.hasNext()) {
      if (nodeDist <= neighborDistances.get(it.next())) {
        radiusList.add(i, node);
        added = true;
      } else {
        i++;
      }
    }
  }

  public void setTarget(List<Double> coords) {
    this.target = coords;
  }

  public List<Coordinate> getResults() {
    return this.radiusList;
  }

  @Override
  public String forApp() {
    return "stars";
  }
}

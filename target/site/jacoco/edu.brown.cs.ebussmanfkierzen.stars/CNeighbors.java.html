package edu.brown.cs.ebussmanfkierzen.stars;

import java.util.Collections;
import java.util.List;
import java.util.ArrayList;

/**
 * CNeighbors class: class dedicated to nearest-neighbors commands
 * with a naive, list-based algorithm.
 */
public class CNeighbors implements MapCommand {

  //true if user is searching for star by name, false otherwise
  private boolean searchByName;
  //coordinates of target star
  private List<Double> target;
  //list of potential nearest neighbors
  private List<Coordinate> candidates;

  public CNeighbors() {
  }


  /**
   * execute: method which finds the n nearest neighbors to a given origin star.
   *
   * @param inputs: an array of search criteria inputted by user
   * @param app:    the application holding the data the user wishes to search through
   * @return A list of strings representing the Star objects that are nearest
   * to the origin and fit the requirements of "inputs"
   **/
  @Override
  public List<String> execute(String[] inputs, Application app) {

    List<String> results = new ArrayList<>();
    this.target = new ArrayList<>();
    //reset search by name (in case of previous command where star was searched for by name
    searchByName = false;

    // Error checking for malformed inputs
    if (!app.isDataLoaded()) {
      System.out.println("ERROR: no loaded data");
      results.add("ERROR: no loaded data");
      return results;
    } else if (!(inputs.length == 5 || inputs.length == 3)) {
      System.out.println("ERROR: incorrect number of arguments for naive_neighbors");
      results.add("ERROR: incorrect number of arguments for naive_neighbors");
      return results;
    } else if (Integer.parseInt(inputs[1]) < 0) {
      System.out.println("ERROR: neighbors must be non-negative");
      results.add("ERROR: neighbors must be non-negative");
      return results;
    }
    if (inputs.length == 5) {
      try {
        target.add(Double.parseDouble(inputs[2]));
        target.add(Double.parseDouble(inputs[3]));
        target.add(Double.parseDouble(inputs[4]));
      } catch (Exception e) {
        System.out.println("ERROR: coordinates must be numeric values");
        results.add("ERROR: coordinates must be numeric values");
        return results;
      }
    } else {
      Coordinate coord = app.findTargetByName(inputs[2]);
      if (coord != null) {
        searchByName = true;
        target = coord.getCoords();
      } else {
        System.out.println("ERROR: cannot find this star by name");
        results.add("ERROR: cannot find this star by name");
        return results;
      }
    }

    //retrieve list of stars in database
    List<Coordinate> itemList = app.getItemList();
    int k = Integer.parseInt(inputs[1]);

    // sort list of stars according to their distance away from origin star
    Collections.sort(itemList, new CoordComparator(app, target));
    // represents candidate stars for final sorted list
    this.candidates = new ArrayList<>();
    // represents stars that are equidistant from origin (in case of ties)
    List<Coordinate> duplicates = new ArrayList<>();
    int numDuplicates = 0;
    int i = 0;
    int duplicateIndex = 0;
    boolean done = false;

    //star was found by name, don't want this star in results
    Coordinate removed = null;
    if (searchByName) {
      // item at top of list should be the star we searched for (closest to itself)
      removed = app.getItemList().get(0);
      app.getItemList().remove(0);
    }

    // so k is an index
    k = k - 1;

    // # of items we want is greater than # of total items, so print all items in list
    if (k >= itemList.size()) {
      candidates = itemList;
      // # of stars we want is less than # of total stars
    } else {
      while (!done) {
        if (i <= k) {

          candidates.add(itemList.get(i));
          // check if last star is same distance from origin as this star to see if we'll need
          // a tiebreaker later
          if (i > 0 && (app.distance(itemList.get(i).getCoords(), target) == app.distance(itemList.get(i - 1).getCoords(), target))) {
            duplicates.add(candidates.get(i));
            if (numDuplicates == 0) {
              duplicateIndex = i;
            }
            numDuplicates++;

            // if consecutive distances aren't equal we don't need to continue keeping track
            // of duplicates
          } else {
            duplicates.clear();
            numDuplicates = 0;
          }
          i++;
          // if we're above k but have non-zero number of running duplicates
        } else if (numDuplicates != 0) {
          // keep searching for duplicates
          while (app.distance(candidates.get(i).getCoords(), target) == app.distance(candidates.get(i - 1).getCoords(), target)) {
            duplicates.add(candidates.get(i));
            numDuplicates++;
            i++;
          }
          done = true;
        } else {
          done = true;
        }
      }

      //past k and no more duplicates-- replace ordered duplicates with random duplicates
      if (numDuplicates > 0) {
        Collections.shuffle(duplicates);
        int index = 0;
        for (int j = duplicateIndex; j <= k; j++) {
          candidates.set(j, duplicates.get(index));
          index++;
        }
      }
    }

    //print the nearest neighbors
    for (Coordinate coord : candidates) {
      System.out.println(coord.getID());
      results.add(coord.toString());
    }

    // need to add a star back to the star list if we removed it earlier because
    // it was searched for my name (so it can be used in future commands)
    if (searchByName) {
      // item at top of list should be the star we searched for (closest to itself)
      itemList.add(removed);
      candidates = itemList.subList(0, itemList.size() - 1);
    }

    return results;
  }

  public void setTarget(List<Double> coords) {
    this.target = coords;
  }

  /**
   * getResults: returns the nearest neighbors as a list of candidates (for testing purposes).
   *
   * @return List of Coordinates
   */
  public List<Coordinate> getResults() {
    return this.candidates;
  }

  @Override
  public String forApp() {
    return "stars";
  }

}

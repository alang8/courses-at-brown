package edu.brown.cs.ebussmanfkierzen.stars;

import java.io.File;
import java.io.IOException;
import java.io.PrintWriter;
import java.io.StringWriter;
import java.util.HashMap;
import java.util.Map;
import java.util.List;

import joptsimple.OptionParser;
import joptsimple.OptionSet;
import spark.ExceptionHandler;
import spark.ModelAndView;
import spark.Request;
import spark.Response;
import spark.Spark;
import spark.QueryParamsMap;
import spark.TemplateViewRoute;
import spark.template.freemarker.FreeMarkerEngine;

import com.google.common.collect.ImmutableMap;

import freemarker.template.Configuration;

/**
 * The Main class of our project. This is where execution begins.
 */
public final class Main {

  private static final int DEFAULT_PORT = 4567;
  private static Application starApp = new StarsApp<>(3);
  private Map<String, Command> commands;

  /**
   * The initial method called when execution begins.
   *
   * @param args An array of command line arguments
   */
  public static void main(String[] args) {
    new Main(args).run();
  }

  private String[] args;

  private Main(String[] args) {
    this.args = args;
  }

  private void run() {
    // Parse command line arguments
    OptionParser parser = new OptionParser();
    parser.accepts("gui");
    parser.accepts("port").withRequiredArg().ofType(Integer.class)
      .defaultsTo(DEFAULT_PORT);
    OptionSet options = parser.parse(args);

    this.starApp = new StarsApp(3);
    Application mapsApp = new MapApp();
    HashMap<String, Application> apps = new HashMap<>();
    apps.put("stars", starApp);
    apps.put("maps", mapsApp);

    if (options.has("gui")) {
      runSparkServer((int) options.valueOf("port"));
    }

    REPL repl = new REPL();
    this.commands = new HashMap<>();
    commands.put("naive_radius", new CRadius());
    commands.put("radius", new CRadius2<Star>());
    commands.put("naive_neighbors", new CNeighbors());
    commands.put("neighbors", new CNeighbors2<Star>());
    commands.put("stars", new CStars());
    commands.put("map", new CMap());
    commands.put("nearest", new CNearest());
    repl.repl(commands, apps);
  }

  private static FreeMarkerEngine createEngine() {
    Configuration config = new Configuration();
    File templates = new File("src/main/resources/spark/template/freemarker");
    try {
      config.setDirectoryForTemplateLoading(templates);
    } catch (IOException ioe) {
      System.out.printf("ERROR: Unable use %s for template loading.%n", templates);
      System.exit(1);
    }
    return new FreeMarkerEngine(config);
  }

  private void runSparkServer(int port) {
    Spark.port(port);
    Spark.externalStaticFileLocation("src/main/resources/static");
    Spark.exception(Exception.class, new ExceptionPrinter());

    FreeMarkerEngine freeMarker = createEngine();

    // Setup Spark Routes
    //first one already here
    Spark.get("/stars", new FrontHandler(), freeMarker);
    Spark.post("/cstars", new StarHandler(), freeMarker);
    Spark.post("/neighbors", new NeighborHandler(), freeMarker);
    Spark.post("/radius", new RadiusHandler(), freeMarker);

  }

  /**
   * Handle requests to the front page of our Stars website.
   */
  private static class FrontHandler implements TemplateViewRoute {
    @Override
    public ModelAndView handle(Request req, Response res) {
      Map<String, Object> variables = ImmutableMap.of("title",
        "Stars: Query the database", "results", "");
      return new ModelAndView(variables, "query.ftl");
    }
  }

  private static class NeighborHandler implements TemplateViewRoute {

    public ModelAndView handle(Request req, Response res) {
      QueryParamsMap qm = req.queryMap();
      String x = qm.value("xval");
      String y = qm.value("yval");
      String z = qm.value("zval");
      String name = qm.value("starName");
      String numNeighbors = qm.value("numNeighbors");
      String naive = qm.value("whichAlgo");
      Command command;
      String[] inputs;
      StringBuilder sb = new StringBuilder(301); // create string builder instance

      if (naive == null || numNeighbors == null || (x == null && y == null && z == null && name == null)) {
        sb.append("ERROR: please fill out all search requirements");
      } else {

        if (naive.equals("naive")) {
          command = new CNeighbors();
        } else {
          command = new CNeighbors2<Star>();
        }
        if (!name.equals("")) {
          inputs = new String[]{"", numNeighbors, name};
        } else {
          inputs = new String[]{"", numNeighbors, x, y, z};
        }

        List<String> results = command.execute(inputs, starApp);
        sb.append("Star ID, Name, Location");
        for (Object item : results) {
          //output = output + "\n" + item;
          sb.append("<br />").append(item);
        }
      }
      Map<String, String> variables = ImmutableMap.of("title", "My Title", "results", sb.toString());
      return new ModelAndView(variables, "query.ftl");
    }
  }

  private static class RadiusHandler implements TemplateViewRoute {

    public ModelAndView handle(Request req, Response res) {
      QueryParamsMap qm = req.queryMap();
      String x = qm.value("xval");
      String y = qm.value("yval");
      String z = qm.value("zval");
      String name = qm.value("starName");
      String radius = qm.value("radius");
      String naive = qm.value("whichAlgo2");
      Command command;
      String[] inputs;
      StringBuilder sb = new StringBuilder(301); // create string builder instance

      if (naive == null || radius == null || (x == null && y == null && z == null && name == null)) {
        sb.append("ERROR: please fill out all search requirements");
      } else {

        if (naive.equals("naive1")) {
          command = new CNeighbors();
        } else {
          command = new CNeighbors2<Star>();
        }
        if (!name.equals("")) {
          inputs = new String[]{"", radius, name};
        } else {
          inputs = new String[]{"", radius, x, y, z};
        }
        List<String> results = command.execute(inputs, starApp);
        sb.append("Star ID, Name, Location");
        for (Object item : results) {
          //output = output + "\n" + item;
          sb.append("<br />").append(item);
        }
      }
      Map<String, String> variables = ImmutableMap.of("title", "My Title", "results", sb.toString());
      return new ModelAndView(variables, "query.ftl");
    }

  }

  private static class StarHandler implements TemplateViewRoute {

    public ModelAndView handle(Request req, Response res) {
      QueryParamsMap qm = req.queryMap();
      String textFromTextField = qm.value("star");
      String[] inputs = new String[]{"", textFromTextField};
      Command command = new CStars();
      List<String> results = command.execute(inputs, starApp);
      String output = "";
      StringBuilder sb = new StringBuilder(301); // create string builder instance

      for (Object item : results) {
        //output = output + "\n" + item;
        sb.append(output).append("<br />").append(item);
      }
      Map<String, String> variables = ImmutableMap.of("title", "My Title", "results", sb.toString());
      return new ModelAndView(variables, "query.ftl");
    }

  }


  /**
   * Display an error page when an exception occurs in the server.
   */
  private static class ExceptionPrinter implements ExceptionHandler {
    @Override
    public void handle(Exception e, Request req, Response res) {
      res.status(500);
      StringWriter stacktrace = new StringWriter();
      try (PrintWriter pw = new PrintWriter(stacktrace)) {
        pw.println("<pre>");
        e.printStackTrace(pw);
        pw.println("</pre>");
      }
      res.body(stacktrace.toString());
    }
  }

}

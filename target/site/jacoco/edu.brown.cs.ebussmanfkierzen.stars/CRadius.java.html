package edu.brown.cs.ebussmanfkierzen.stars;

import java.util.Collections;
import java.util.List;
import java.util.ArrayList;

/**
 * CRadius class: class used for functionality related
 * to finding items within a specified radius of a target.
 */
public class CRadius implements Command {

  private List<String> results;
  private boolean searchByName;
  private Application app;
  private List<Double> target;
  private List<Coordinate> candidates;

  public CRadius() {
  }

  /**
   * execute: finds the specified number of elements within a given radius of a target
   * using a naive, list-based algorithm.
   *
   * @param inputs the search criteria for the radius search (name, radius value, etc)
   * @param appl   the application holding the data the user wishes to search through
   * @return a list of strings representing items within a given radius of a target
   */
  public List<String> execute(String[] inputs, Application appl) {

    this.app = appl;
    results = new ArrayList<>();
    //reset search by name
    searchByName = false;
    this.target = new ArrayList<>();
    // do all error-checking inside this function
    if (!app.isDataLoaded()) {
      System.out.println("ERROR: no loaded data");
      results.add("ERROR: no loaded data");
      return results;
    } else if (!(inputs.length == 5 || inputs.length == 3)) {
      System.out.println("ERROR: incorrect number of arguments for naive_radius");
      results.add("ERROR: incorrect number of arguments for naive_radius");
      return results;
    } else if (Integer.parseInt(inputs[1]) < 0) {
      System.out.println("ERROR: radius must be non-negative");
      results.add("ERROR: radius must be non-negative");
      return results;
    }
    if (inputs.length == 5) {
      try {
        target.add(Double.parseDouble(inputs[2]));
        target.add(Double.parseDouble(inputs[3]));
        target.add(Double.parseDouble(inputs[4]));
      } catch (Exception e) {
        System.out.println("ERROR: coordinates must be numeric values");
        results.add("ERROR: coordinates must be numeric values");
        return results;
      }
    } else {
      Coordinate coord = app.findTargetByName(inputs[2]);
      if (coord != null) {
        searchByName = true;
        target = coord.getCoords();
      } else {
        System.out.println("ERROR: cannot find this star by name");
        results.add("ERROR: cannot find this star by name");
        return results;
      }
    }

    // sort list of stars according to their distance away from origin star
    //Collections.sort(app.getItemList(), new CoordComparator(app));
    Collections.sort(app.getItemList(), new CoordComparator(app, target));
    Coordinate removed = null;

    if (searchByName) {
      // item at top of list should be the star we searched for (closest to itself)
      removed = app.getItemList().get(0);
      app.getItemList().remove(0);
    }

    //int radius = Integer.parseInt(app.getArgs()[1]);
    int radius = Integer.parseInt(inputs[1]);
    int i = 0;
    this.candidates = new ArrayList<>();

    // add all stars less than the given radius away from our sorted list
    while (i < app.getItemList().size()) {
      if (app.distance(app.getItemList().get(i).getCoords(), target) <= radius) {
        candidates.add(app.getItemList().get(i));
        i++;
      } else {
        break;
      }
    }

    List<String> result = new ArrayList<>();

    for (Coordinate coord : candidates) {
      System.out.println(coord.getID());
      result.add(coord.toString());
    }

    // need to add a star back to the star list if we removed it earlier because
    // it was searched for my name (so it can be used in future commands)
    //if (starApp.getOriginName() != null) {
    if (searchByName) {
      // item at top of list should be the star we searched for (closest to itself)
      //make sure this will actually add it back to the starApp list
      app.getItemList().add(removed);
    }

    return result;

  }

  /**
   * setTarget: sets the object that will serve as the center of the radius search.
   *
   * @param coords a list of coordinates representing the target point
   */
  public void setTarget(List<Double> coords) {
    this.target = coords;
  }

  /**
   * getResults: returns the nearest neighbors as a list of candidates (for testing purposes).
   *
   * @return List of Coordinates
   */
  public List<Coordinate> getResults() {
    return this.candidates;
  }

  @Override
  public String forApp() {
    return "stars";
  }
}

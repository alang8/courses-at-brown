package edu.brown.cs.ebussmanfkierzen.stars;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.regex.Pattern;
import java.util.regex.MatchResult;
import java.util.Map;

/**
 * class that contains a Read-Evaluation-Print loop.
 */
public class REPL {

  /**
   * repl method: given command options and an application, takes in user input
   * and executes commands if possible.
   *
   * @param commands: a map from the string representation of a command to a Command instance
   *                  capable of executing the corresponding command.
   * @param apps:      a map from string rep of an app to an App instance used for executing the command.
   */
  public void repl(Map<String, Command> commands, Map<String, Application> apps) {

    BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
    String line;

    while ((line = readline(in)) != null) {

      //parse inputs for quotation marks and names of stars
      String[] inputs = Pattern.compile("[^\\s\"']+|\"([^\"]*)\"|'([^']*)'")
        .matcher(line)
        .results()
        .map(MatchResult::group)
        .toArray(String[]::new);

      if (inputs.length > 0) {
        System.out.println("HERE!");
        try {
          System.out.println("R1");
          Command c = commands.get(inputs[0]);
          System.out.println("R2");
          Application app = apps.get(c.forApp());
          System.out.println("R3");
          c.execute(inputs, app);
        } catch (Exception e) {
          System.out.println("ERROR: improper command");
        }
      }
    }
  }

  /**
   * readLine: reads one line of text, given a buffered reader.
   *
   * @param in: reader through which to read text
   * @return String of read text
   */
  private static String readline(BufferedReader in) {
    String line = "";
    try {
      line = in.readLine();
    } catch (IOException e) {
      System.out.println("IOException: cannot read input");
    }
    return line;
  }
}

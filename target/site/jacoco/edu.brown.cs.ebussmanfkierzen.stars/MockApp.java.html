package edu.brown.cs.ebussmanfkierzen.stars;

import java.util.List;

public class MockApp {

  public MockApp() {
  }

  /**
   * execute method: handles commands related to mock persons.
   *
   * @param input a list of Strings representing components of command-line input
   */
  public void execute(String[] input) {
    if (input.length != 2) {
      System.out.println("ERROR: incorrect number of arguments for mock");
    } else if (input[0].equals("mock")) {
      parseCSV(input[1]);
    }
  }

  /**
   * parseCSV method: handles parsing of CSV files related to human data.
   *
   * @param filename the name of the CSV file to be parsed
   */
  public void parseCSV(String filename) {

    CSVParse parser = new CSVParse();
    List<String[]> personList = parser.csvParse(filename);

    for (String[] item : personList) {
      MockPerson newPerson = new MockPerson(item[0], item[1], item[2], item[3], item[4], item[5]);
      System.out.println(newPerson.toString());
    }
  }

}

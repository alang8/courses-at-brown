package edu.brown.cs.ebussmanfkierzen.stars;


import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;
import java.util.PriorityQueue;
import java.util.Collections;

/**
 * Class representing the nearest command.
 */
public class CNearest implements Command {
  private List<String> results = new ArrayList<>();
  private List<Coordinate> nearest = new ArrayList<>();

  @Override
  public List<String> execute(String[] inputs, Application app) {
    if (inputs.length != 3) {
      System.out.println("ERROR: Nearest command expects 2 numerical arguments.");
      results.add("ERROR: Nearest command expects 2 numerical arguments.");
    } else {
      try {
        double lat = Double.parseDouble(inputs[1]);
        double lon = Double.parseDouble(inputs[2]);
        List<Double> source = new ArrayList<>();
        source.add(lat);
        source.add(lon);
        Comparator<Coordinate> myDoubleComp = Comparator.comparingDouble(s -> s.distanceFrom(source));
        Comparator<Coordinate> nodeComp = Collections.reverseOrder(myDoubleComp);
        PriorityQueue<Coordinate> pq = new PriorityQueue<>(1, nodeComp);
        KDNode.kdnAlg(pq, app.getData(), source, 0, 1);
        Node nearestNode = (Node) pq.peek();
        nearest.add(nearestNode);
        System.out.println(nearestNode.getName());
        results.add(nearestNode.getName());
      } catch (NumberFormatException e) {
        System.out.println("ERROR: Nearest command expects 2 numerical arguments.");
        results.add("ERROR: Nearest command expects 2 numerical arguments.");
      }
    }
    return results;
  }

  @Override
  public void setTarget(List<Double> target) {
    return;
  }

  @Override
  public List<Coordinate> getResults() {
    return nearest;
  }

  @Override
  public String forApp() {
    return "maps";
  }
}

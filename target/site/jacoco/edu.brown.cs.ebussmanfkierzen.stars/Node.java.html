package edu.brown.cs.ebussmanfkierzen.stars;

import java.util.ArrayList;
import java.util.List;

public class Node implements Coordinate {
  private static final double RADIUS = 6371.0;
  private ArrayList<Double> coords = new ArrayList<>();
  private String id;

  public Node(String id, double lat, double lon) {
    this.coords.add(lat);
    this.coords.add(lon);
    this.id = id;
  }

  @Override
  public ArrayList<Double> getCoords() {
    return coords;
  }

  @Override
  public int getID() {
    return Integer.parseInt(id.substring(3).replaceAll("[.]", ""));
  }

  @Override
  public String getName() {
    return this.id;
  }

  /**
   * Method which returns the haversine distance from another node.
   * @param other - the other node to measure the distance from.
   * @return - the distance from that node.
   */
  public double distanceFrom(List<Double> other) {
    double sum = 0;
    for (int i = 0; i < other.size(); i++) {
      sum += Math.pow((this.coords.get(i) - other.get(i)), 2);
    }
    return Math.sqrt(sum);
  }

  /**
   *  Method which returns the haversine distance b/t two points.
   * @param other - the lat and longitude of another Node.
   * @return - the haversine distnace between this node and other.
   */
  public double haversineDistance(List<Double> other) {
    double lat1 = this.coords.get(0);
    double lon1 = this.coords.get(1);
    double lat2 = other.get(0);
    double lon2 = other.get(1);
    double avgLat = (lat2 - lat1) / 2;
    double avgLon = (lon2 - lon1) / 2;
    double dis = Math.sin(avgLat) + Math.cos(lat1) * Math.cos(lat2) * Math.pow(Math.sin(avgLon), 2);
    dis = Math.sqrt(dis);
    dis = 2 * RADIUS * Math.asin(dis);
    return dis;
  }

  public String toString() {
    return this.getName();
  }


}

package edu.brown.cs.ebussmanfkierzen.stars;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.List;

/**
 * Class representing the map command.
 */
public class CMap implements Command {
  private MapApp app;
  private List<String> results = new ArrayList<>();

  @Override
  public List<String> execute(String[] inputs, Application ap) {
    this.app = (MapApp) ap;
    if (inputs.length != 2) {
      System.out.println("ERROR: map expects 1 argument, recieved " + (inputs.length - 1));
      results.add("ERROR: map expects 1 argument, recieved " + (inputs.length - 1));
    } else {
      try {
        this.app.setPathToDB(inputs[1]);
        createTree(inputs[1]);
        System.out.println("map set to " + inputs[1]);
        results.add("map set to " + inputs[1]);
      } catch (SQLException throwables) {
        System.out.println("ERROR: Invalid SQL File");
        results.add("ERROR: Invalid SQL File");
      } catch (ClassNotFoundException e) {
        System.out.println("ERROR: Invalid SQL File");
        results.add("ERROR: Invalid SQL File");
      }
    }
    return results;
  }

  /**
   * Method which creates a KD tree from the nodes in the map.
   * @param pathToDB - the path to the database.
   * @throws SQLException
   */
  private void createTree(String pathToDB) throws SQLException {
    //build tree from list
    List<Node> nodeList = new ArrayList<>();

    Connection conn = this.app.getConn();

    //select traversable nodes
    PreparedStatement prep = conn.prepareStatement(
      "SELECT * FROM Node WHERE id IN (SELECT start FROM Way) OR id IN (SELECT end FROM Way);"
    );
    ResultSet rs = prep.executeQuery();
    while (rs.next()) {
      String id = rs.getString(1);
      double lat = rs.getDouble(2);
      double lon = rs.getDouble(3);
      nodeList.add(new Node(id, lat, lon));
    }

    rs.close();
    prep.close();

    IKDTree mapTree = KDNode.buildTree(nodeList, 2, 0);

    app.setData(mapTree);
    app.setDataLoaded();
  }

  @Override
  public void setTarget(List<Double> target) { }

  @Override
  public List<Coordinate> getResults() {
    return null;
  }

  @Override
  public String forApp() {
    return "maps";
  }
}

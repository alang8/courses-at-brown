package edu.brown.cs.ebussmanfkierzen.stars;

import java.util.List;
import java.util.ArrayList;

/**
 * Star: class representing a 3-dimensional star object.
 */
public class Star implements Coordinate {

  private final Integer id;
  private final String name;
  private final double x;
  private final double y;
  private final double z;
  private final List<Double> coords;

  public Star(int id, String name, double x, double y, double z) {
    this.id = id;
    this.name = name;
    this.x = x;
    this.y = y;
    this.z = z;
    this.coords = new ArrayList<>();
    this.coords.add(this.x); //index 0 (depth 0)
    this.coords.add(this.y); //index 1 (depth 1)
    this.coords.add(this.z); //index 2 (depth 2)
  }

  public Star(double x, double y, double z) {
    this.x = x;
    this.y = y;
    this.z = z;
    this.coords = new ArrayList<>();
    this.coords.add(this.x); //index 0 (depth 0)
    this.coords.add(this.y); //index 1 (depth 1)
    this.coords.add(this.z); //index 2 (depth 2)
    this.id = null;
    this.name = "none";
  }

  public List<Double> getCoords() {
    return this.coords;
  }


  public int getID() {
    return this.id;
  }

  public String getName() {
    return this.name;
  }

  @Override
  public double distanceFrom(List<Double> crds) {
    double sum = 0;
    for (int i = 0; i < crds.size(); i++) {
      sum += Math.pow((this.coords.get(i) - crds.get(i)), 2);
    }
    return Math.sqrt(sum);
  }

  @Override
  public String toString() {
    return this.id + " " + this.name + " " + "(" + this.x + ", " + this.y + ", " + this.z + ") ";
  }

  @Override
  public boolean equals(Object o) {
    if (o == this) {
      return true;
    }

    if (!(o instanceof Star)) {
      return false;
    }

    Star other = (Star) o;

    boolean nameEQ = this.getName().equals(other.getName());
    boolean checkX = this.x == other.x;
    boolean checkY = this.y == other.y;
    boolean checkZ = this.z == other.z;
    boolean coorEQ = checkX && checkY && checkZ;
    boolean idEQ = this.getID() == other.getID();
    return nameEQ && coorEQ && idEQ;
  }

  @Override
  public int hashCode() {
    return this.getID();
  }


}
